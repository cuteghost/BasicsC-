/*Zadatak: Napišite program, poštujuci sve faze procesa programiranja,  u kojem cete za uneseni prirodni broj n izracunati sumu:
S=1!/3^2 +2!/5^2 +…+n!/?(2n+1)?^2

Upotrijebite funkcije:
float suma (int);

float suma(int);
int faktorijel(int);
int main()
{
	int broj;
	cin >> broj;
	cout << "rezultat je -> " << suma(broj);
	system("pause>0");
}
float suma(int broj)
{
	float rez = 0;
	for (int i = 1; i <= broj; i++)
	{
		rez += (faktorijel(i)) / (pow((2 * i + 1), 2));
	}
	return rez;
}
int faktorijel(int broj)
{
	int rez = 1;
	for (int i = 1; i <= broj; i++)
	{
		rez *= i;
	}
	return rez;
}
