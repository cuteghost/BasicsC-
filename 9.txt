/* Zadatak: Napišite program, , poštujuci sve faze procesa programiranja, koji omogucava unos prirodnog broja n preko tastature te izracunava sumu:

Upotrijebite funkcije:

Unos prirodnog broja i ispis ukupne sume vršiti u funkciji main.

int faktorijel(int);
void suma(int, float &);
float ukupnasuma(int);

int main()
{
	int n;
	cout << "Unesite broj ---> "; cin >> n;
	cout << "Rezultat je ---> " << ukupnasuma(n) << endl;
	system("pause>0");

}
int faktorijel(int broj)
{
	int rez = 1;
	for (int i = 1; i <=broj; i++)
	{
		rez *= i;
	}
	return rez;
}

float ukupnasuma(int broj)
{
	float rezultat = 0, sumare = 0;

	for (int i = 1; i <=broj; i++)
	{
		suma(i, sumare);
		rezultat += (float)(faktorijel(i)) / sumare;
	}
	return rezultat;
}
void suma(int broj, float & nesto)
{
		nesto += ((float)1 / (float)(broj+1));	
}
